for pixel in canvas:
    ray = construct_ray(eye, pixel)
    
    closest = None
    for object in scene:
        if (ray.hits(object) and 
            (closest == None or
             distance(object, eye) <
             distance(closest, eye))):
            closest = object
    
            do_shading(closest, ray)
